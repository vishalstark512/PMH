%% ── Shared pgfplots styles ──────────────────────────────────────────────────
\pgfplotsset{
  pmh base/.style={
    width=0.92\linewidth, height=5.5cm,
    tick label style={font=\small},
    label style={font=\small},
    legend style={font=\footnotesize, draw=none, fill=none,
                  at={(0.02,0.98)}, anchor=north west},
    grid=both,
    grid style={line width=0.3pt, draw=gray!20},
    minor grid style={draw=none},
    line width=0.8pt,
    mark size=1.8pt,
  },
  pmh bar/.style={
    pmh base,
    ybar, bar width=5pt,
    enlarge x limits=0.15,
    xtick align=inside,
  },
  pmh hbar/.style={
    pmh base,
    xbar, bar width=5pt,
    enlarge y limits=0.15,
  },
}
